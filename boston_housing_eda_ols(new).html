## Step 6.1: Finding MSE for Simple Regression
# Load the tools and data
import pandas as pd
import statsmodels.api as sm

# Load the dataset
url = "https://raw.githubusercontent.com/selva86/datasets/master/BostonHousing.csv"
data = pd.read_csv(url)
data.columns = data.columns.str.upper()

# Define predictors (all columns except CRIM)
predictors = [col for col in data.columns if col != 'CRIM']

# Calculate MSE for each simple regression model
print("Calculating MSE for each simple regression model:")
y = data['CRIM']  # Real CRIM values
for predictor in predictors:
    X = data[[predictor]]
    X = sm.add_constant(X)
    model = sm.OLS(y, X).fit()
    predictions = model.predict(X)  # Model's guesses for CRIM
    mse = ((y - predictions) ** 2).mean()  # Calculate MSE
    print(f"MSE for {predictor}: {mse:.2f}")
